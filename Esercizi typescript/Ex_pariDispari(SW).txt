/*
Si scriva un programma che preso in input un valore numerico, verifichi se questo sia PARI (EVEN) o DISPARI (ODD)
a schermo :
Input: 2
Output:  "Il numero e` pari "
Input: 3
"Il numero e` dispari "
*/

let numero:number=prompt("Inserisci un numero")*1;

switch (numero%2) {
    case 0: {console.log("Il numero è pari");break}
    default: {console.log("Il numero è dispari");break}
}
